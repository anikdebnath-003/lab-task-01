#include <windows.h>  // for MS Windows
#include <GL/glut.h>  // GLUT, include glu.h and gl.h

void initGL() {
    // Set background color to white
    glClearColor(1.0f, 1.0f, 1.0f, 1.0f);
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);   // Clear the color buffer

    
    glLineWidth(2.0f);
    glBegin(GL_LINES);
        glColor3f(0.0f, 0.0f, 0.0f); // Black color

       
        glVertex2f(-1.0f, 0.0f);
        glVertex2f( 1.0f, 0.0f);

        glVertex2f(0.0f,  1.0f);
        glVertex2f(0.0f, -1.0f);
    glEnd();

   
    glBegin(GL_QUADS);
        glColor3f(1.0f, 0.0f, 0.0f); // Red
        glVertex2f(-0.6f, 0.2f);   
        glVertex2f(-0.1f, 0.2f);   
        glVertex2f(-0.1f, 0.6f);   
        glVertex2f(-0.6f, 0.6f);   
    glEnd();

    // ---- Quadrant 1 (top-right): Green Arrow ----
    // Arrow Body using GL_QUADS
    glBegin(GL_QUADS);
        glColor3f(0.0f, 0.6f, 0.0f); // Green
        glVertex2f(0.1f,  0.35f);  // Bottom-left of body
        glVertex2f(0.4f,  0.35f);  // Bottom-right of body
        glVertex2f(0.4f,  0.55f);  // Top-right of body
        glVertex2f(0.1f,  0.55f);  // Top-left of body
    glEnd();

    // Arrow Head using GL_TRIANGLES
    glBegin(GL_TRIANGLES);
        glColor3f(0.0f, 0.6f, 0.0f); // Green
        glVertex2f(0.4f,  0.25f);  // Bottom point of triangle
        glVertex2f(0.4f,  0.65f);  // Top point of triangle
        glVertex2f(0.65f, 0.45f);  // Right tip (pointing right)
    glEnd();

    // ---- Quadrant 3 (bottom-left): Purple Triangle (pointing left) ----
    glBegin(GL_TRIANGLES);
        glColor3f(0.5f, 0.0f, 0.5f); // Purple
        glVertex2f(-0.1f, -0.2f);  // Top-right
        glVertex2f(-0.1f, -0.6f);  // Bottom-right
        glVertex2f(-0.5f, -0.4f);  // Left tip (pointing left)
    glEnd();

    // ---- Quadrant 4 (bottom-right): Yellow Triangle ----
    glBegin(GL_TRIANGLES);
        glColor3f(1.0f, 1.0f, 0.0f); // Yellow
        glVertex2f(0.2f, -0.6f);   // Bottom-left
        glVertex2f(0.6f, -0.6f);   // Bottom-right
        glVertex2f(0.4f, -0.2f);   // Top (tip)
    glEnd();

    glFlush();  // Render now
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);                        // Initialize GLUT
    glutCreateWindow("4 Objects in 4 Quadrants"); // Create window with title
    glutInitWindowSize(500, 500);                 // Set window size
    glutInitWindowPosition(50, 50);               // Set window position
    glutDisplayFunc(display);                     // Register display callback
    initGL();                                     // Initialize OpenGL settings
    glutMainLoop();                               // Enter event-processing loop
    return 0;
}
